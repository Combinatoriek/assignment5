$a_n = An + B$

$a_{n - 1} = A(n - 1) + B$

$\begin{aligned}
	An + B &= 2(A(n - 1) + B) + n + 5\\
	&= 2A(n - 1) + 2B + n + 5\\
	&= 2An - 2A + 2B + n + 5\\
	&= 2An + n -2A + 2B + 5
\end{aligned}$

$\begin{aligned}
	2An + n &= An\\
	n(2A + 1) &= An\\
	2A + 1 &= A\\
	A + 1 &= 0\\
	A &= -1
\end{aligned}$

$\begin{aligned}
	-2A + 2B + 5 &= B\\
	-2(-1) + 2B + 5 &= B\\
	2 + 2B + 5 &= B\\
	B + 7 &= 0\\
	B &= - 7
\end{aligned}$

$a_n = -n -7$
